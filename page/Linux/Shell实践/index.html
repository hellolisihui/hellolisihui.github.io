Shell 脚本的最佳实践

#!/usr/bin/env bash

1. 定义变量

# export DAT_PATH='/usr/local/nginx'
DAT_PATH=${DAT_PATH:-'/usr/local/nginx'}

2. 检查是否以 root 身份运行

check_if_running_as_root() {
  if [[ "$UID" -ne '0' ]]; then
    echo "WARNING: The user currently executing this script is not root."
    read -r -p "Are you sure you want to continue? [y/n] " enter_information
    if [[ "${enter_information:0:1}" = 'y' ]]; then
      echo "Continuing the installation with current user..."
    else
      echo "Not running with root, exiting..."
      exit 1
    fi
  fi
}
check_if_running_as_root

3. 输出文本高亮显示

main() {
  red=$(tput setaf 1)
  green=$(tput setaf 2)
  yellow=$(tput setaf 3)
  blue=$(tput setaf 4)
  magenta=$(tput setaf 5)
  aoi=$(tput setaf 6)
  white=$(tput setaf 7)
  reset=$(tput sgr0)
  blink=$(tput blink)
  bold=$(tput bold)
  smso=$(tput smso)
  smul=$(tput smul) 

  echo "${red}WARNING${reset}红色"
  echo "${green}WARNING${reset}绿色"
  echo "${aoi}WARNING${reset}青色"
  echo "${yellow}WARNING${reset}黄色"
  echo "${blue}WARNING${reset}蓝色"
  echo "${magenta}WARNING${reset}洋红"
  echo "${white}WARNING${reset}白色"
  echo "${blink}WARNING${reset}闪烁"
  echo "${bold}WARNING${reset}粗体"
  echo "${smso}WARNING${reset}突出"
  echo "${smul}WARNING${reset}下划线"
}
main

4. 打印菜单

menu() {
cat <<EOF
1) 查看当前用户
2) 查看主机名称
3) 查看环境变量
EOF
read -p "输入序号:" num
case $num in
  1)
  echo $USER
  ;;
  2)
  echo $HOSTNAME
  ;;
  3)
  echo $PATH
  ;;
  *)
  echo '*'
esac
}
menu
